this is continiuing from my last tutorial, open your bot's index.js file,

(THE client.login('TOKEN') MUST ALWAYS BE IN THE BOTTOM NOTHING SHOULD BE WRITTEN UNDER IT!)

now over your bot token login you can type:

client.on('message', message =>{
if(!message.content.startsWith(prefix) || message.author.bot) return;

const args = message.content.slice(prefix.length).split(/ +/);
const command = args.shift().toLowerCase();
})

this is the template for you bot, now inside the {}'s you can type:

if(command === 'test'){
message.channel.send('hello world!');
}

const Discord = require('discord.js'); const prefix = 'your prefix here :)'; const client = new Discord.Client(); client.once('ready', () => {
